from getpass import getpass

Paths & Config#

HOST = 'localhost'
DATABASE = 'algojungle'

Connection to database#

PASSWORD = getpass("Enter Password: ")

conn = psycopg2.connect(host=HOST, 
                        database=DATABASE, 
                        user=USER, 
                        password=PASSWORD)
